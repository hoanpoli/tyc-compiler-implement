		- Indexing and Slicing

			slicing: x[start:end:step]
					x[1:3] => [2, 'a']
					x[1:] => [2, 'a', 'c', 4.3]
					x[:-1] => [1, 2, 'a', 'c']
					x[:] => [1, 2, 'a', 'c', 4.3] đây là 1 list mới
					x[1:4:2] => [2, 'c']
					x[::2] => [1, 'a', 4.3]
					x[::-1] => [4.3, 'c', 'a', 2, 1]
